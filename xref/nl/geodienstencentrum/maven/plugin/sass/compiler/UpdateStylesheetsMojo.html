20  package nl.geodienstencentrum.maven.plugin.sass.compiler;
21  
22  import static org.apache.maven.plugins.annotations.LifecyclePhase.PROCESS_SOURCES;
23  import nl.geodienstencentrum.maven.plugin.sass.AbstractSassMojo;
24  
25  import org.apache.maven.plugin.MojoExecutionException;
26  import org.apache.maven.plugin.MojoFailureException;
27  import org.apache.maven.plugins.annotations.Mojo;
28  
29  /**
30   * Mojo that compiles Sass sources into CSS files using
31   * {@code update_stylesheets}.
32   */
33  @Mojo(name = "update-stylesheets", defaultPhase = PROCESS_SOURCES)
34  public class UpdateStylesheetsMojo extends AbstractSassMojo {
35  
36  	/**
37  	 * Execute the compiler script.
38  	 *
39  	 * @see org.apache.maven.plugin.Mojo#execute()
40  	 * @throws MojoExecutionException when the execution of the plugin
41  	 *         errored
42  	 * @throws MojoFailureException when the sass compilation fails
43  	 *
44  	 */
45  	@Override
46  	public void execute() throws MojoExecutionException, MojoFailureException {
47  		this.getLog().info("Compiling Sass Templates");
48  
49  		// build sass script
50  		final StringBuilder sassBuilder = new StringBuilder();
51  		this.buildBasicSassScript(sassBuilder);
52  		sassBuilder.append("Sass::Plugin.update_stylesheets");
53  		final String sassScript = sassBuilder.toString();
54  
55  		// ...and execute
56  		this.executeSassScript(sassScript);
57  	}
58  }
